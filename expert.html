    /* future.js */

    /* askBeforeSubmitPlugin.js */
    $.fn.askBeforeSubmit = function () {
        $(this).on("click", function () {
            return confirm("Are You Sure?");
        });
    };

    /* main.js */
    $(function () {
        var source, template,
            model = {
                buttonValue: "Send",
                formAction: "success.html"
            };

        //render template
        source   = $("#template").html();
        template = Handlebars.compile(source);
        $("#container").html(template(model));

        //attach plugin
        $("#button").askBeforeSubmit();
    })
